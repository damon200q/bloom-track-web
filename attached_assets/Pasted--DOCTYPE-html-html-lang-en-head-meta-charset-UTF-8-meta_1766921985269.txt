        // Due Date Calculator
        function calculateDueDate() {
            const method = document.querySelector('input[name="dueMethod"]:checked').value;
            let lmpDate;

            if (method === 'lmp') {
                const lmpInput = document.getElementById('lmpDate').value;
                if (!lmpInput) {
                    alert("Please enter your last period date");
                    return;
                }
                lmpDate = new Date(lmpInput);
            } else {
                const conceptionInput = document.getElementById('conceptionDate').value;
                if (!conceptionInput) {
                    alert("Please enter your conception date");
                    return;
                }
                const conceptionDate = new Date(conceptionInput);
                lmpDate = new Date(conceptionDate);
                lmpDate.setDate(conceptionDate.getDate() - 14);
            }

            // Naegele's Rule
            const dueDate = new Date(lmpDate);
            dueDate.setDate(lmpDate.getDate() + 280);

            const today = new Date();
            const daysDiff = Math.floor((today - lmpDate) / (1000 * 60 * 60 * 24));
            const weeks = Math.floor(daysDiff / 7);
            const days = daysDiff % 7;

            let trimester = "First Trimester";
            if (weeks >= 27) trimester = "Third Trimester";
            else if (weeks >= 13) trimester = "Second Trimester";

            const options = { year: 'numeric', month: 'long', day: 'numeric' };
            document.getElementById('dueDate').innerText = dueDate.toLocaleDateString('en-US', options);
            document.getElementById('currentWeek').innerText = `Week ${weeks}, Day ${days}`;
            document.getElementById('daysPregnant').innerText = `${daysDiff} days pregnant`;
            document.getElementById('trimester').innerText = trimester;

            document.getElementById('dueDateResult').classList.remove('hidden');
        }

        // Weight Calculator
        function calculateWeight() {
            const weight = parseFloat(document.getElementById('preWeight').value);
            const height = parseFloat(document.getElementById('height').value);
            const currentWeek = parseInt(document.getElementById('currentWeekInput').value) || 0;

            if (!weight || !height) {
                alert("Please enter your weight and height");
                return;
            }

            const heightM = height / 100;
            const bmi = weight / (heightM * heightM);
            
            let category = "";
            let minGain = 0, maxGain = 0;

            if (bmi < 18.5) {
                category = "Underweight";
                minGain = 12.5;
                maxGain = 18;
            } else if (bmi < 25) {
                category = "Normal weight";
                minGain = 11.5;
                maxGain = 16;
            } else if (bmi < 30) {
                category = "Overweight";
                minGain = 7;
                maxGain = 11.5;
            } else {
                category = "Obese";
                minGain = 5;
                maxGain = 9;
            }

            document.getElementById('bmiValue').innerText = bmi.toFixed(1);
            document.getElementById('bmiCategory').innerText = category;
            document.getElementById('weightRange').innerText = `${minGain} - ${maxGain} kg throughout pregnancy`;

            if (currentWeek > 0) {
                let expectedGain = 0;
                if (currentWeek <= 12) {
                    expectedGain = 1.5;
                } else {
                    const weeklyRate = (minGain + maxGain) / 2 / 28;
                    expectedGain = 1.5 + (currentWeek - 12) * weeklyRate;
                }
                document.getElementById('weeklyGain').innerText = `Approximately ${expectedGain.toFixed(1)} kg by week ${currentWeek}`;
                document.getElementById('currentWeekWeight').classList.remove('hidden');
            }

            document.getElementById('weightResult').classList.remove('hidden');
        }
